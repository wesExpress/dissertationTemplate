\usepackage{titlesec}
\titleformat*{\section}{\fontsize{12}{15}\bfseries}

% makes chapter titles 12 pt font
\titleformat{\chapter}[display]
{\normalfont \bfseries\centering}
{\chaptertitlename\ \thechapter}{12pt}
{\normalfont \bfseries\centering}

\hypersetup{

    citecolor=black,
    filecolor=black,
    linkcolor=black,
    urlcolor=black,
    linktoc=all
}
